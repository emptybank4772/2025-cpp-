#include<iostream>
using namespace std;
class Shape {
public:
	virtual double area() = 0;
};
class Circle : public Shape {
private:
	double radius;
public:
	Circle(double a) : radius(a) {}
	double area() override {
		return 3.14 * radius * radius;
	}
};
class Rectangle : public Shape {
private:
	int w, h;
public:
	Rectangle(int a, int b) : w(a), h(b) {}
	double area() override {
		return w * h;
	}
};
int main() {
	Circle c(3);
	Rectangle r(3, 4);
	cout << c.area() <<endl;
	cout << r.area();
}

#include<iostream>
template<typename T>
void swapValuse(T& a, T& b) {
	T temp = a;
	a = b;
	b = temp;
}
int main() {
	int a = 10, b = 20;
	std::cout << a << " " << b << std::endl;
	swapValuse(a, b);
	std::cout << a << " " << b << std::endl;
}
